// cart_provider.dart
Map<int, List<CartItem>> get groupedBySupplier {
  final Map<int, List<CartItem>> grouped = {};
  for (var item in _cartItems) {
    if (!grouped.containsKey(item.adminId)) {
      grouped[item.adminId] = [];
    }
    grouped[item.adminId]!.add(item);
  }
  return grouped;
}

// routes/cart.js - GET /
const groupedItems = {};
cartItems.forEach(item => {
  if (!groupedItems[item.admin_id]) {
    groupedItems[item.admin_id] = {
      admin_id: item.admin_id,
      admin_name: item.admin_name,
      company_name: item.company_name,
      items: [],
      subtotal: 0
    };
  }
  groupedItems[item.admin_id].items.push(item);
  groupedItems[item.admin_id].subtotal += item.price * item.quantity;
});
